#include <stdio.h>
#include <stdlib.h>

typedef struct Edge {
    int weight;
    int destination;
    struct Edge* next;
} Edge;

typedef struct Node {
    int vertex;
    Edge* head;
} Node;

typedef struct Graph {
    int vertices;
    Node* adjList;
} Graph;

void addEdge(Graph* graph, int src, int dest, int weight) {
    Edge* newEdge = (Edge*)malloc(sizeof(Edge));
    newEdge->destination = dest;
    newEdge->weight = weight;
    newEdge->next = graph->adjList[src].head;
    graph->adjList[src].head = newEdge;

    // 若是是无向图，需要在目标节点处也添加一条边
    // newEdge = (Edge*)malloc(sizeof(Edge));
    // newEdge->destination = src;
    // newEdge->weight = weight;
    // newEdge->next = graph->adjList[dest].head;
    // graph->adjList[dest].head = newEdge;
}

void printGraph(Graph* graph) {
    for (int i = 0; i < graph->vertices; i++) {
        printf("Vertex %d:", i);
        Edge* currentEdge = graph->adjList[i].head;
        while (currentEdge != NULL) {
            printf(" -> %d(%d)", currentEdge->destination, currentEdge->weight);
            currentEdge = currentEdge->next;
        }
        printf("\n");
    }
}

void initGraph() {
    Graph graph;

    printf("Enter the number of vertices in the directed graph: ");
    scanf("%d", &graph.vertices);
    graph.array = (Node*)malloc(graph.vertices * sizeof(Node));

    for (int i = 0; i < graph.vertices; i++) {
        graph.array[i].vertex = i;
        graph.array[i].head = NULL;
    }

    int numEdges;
    printf("Enter the number of directed edges: ");
    scanf("%d", &numEdges);
    for (int i = 0; i < numEdges; i++) {
        int src, dest;
        printf("Enter directed edge (source destination): ");
        scanf("%d %d", &src, &dest);
        addDirectedEdge(&graph, src, dest);
    }

    printf("Directed Graph:\n");
    printDirectedGraph(&graph);
}

Graph buildReverseGraph(Graph* graph) {
    Graph reverseGraph;
    reverseGraph.vertices = graph->vertices;
    reverseGraph.adjList = (Node*)malloc(sizeof(Node) * graph->vertices);
    
    for (int i = 0; i < graph->vertices; ++i) {
        reverseGraph.adjList[i].vertex = graph->adjList[i].vertex;
        reverseGraph.adjList[i].head = NULL;
        
        Edge* currentEdge = graph->adjList[i].head;
        while (currentEdge != NULL) {
            // Add reverse edge to the new graph
            addEdge(&reverseGraph, currentEdge->destination, graph->adjList[i].vertex, weight);
            currentEdge = currentEdge->next;
        }
    }
    return reverseGraph;
}

// 计算出度；
void getDegree(Graph graph, int* degreeArray) {
    for (int i = 0; i < graph.vertices; ++i) {
        degreeArray[i] = 0;  // 初始化每个顶点的度为零

        Edge* edge = graph.adjList[i].head;
        int vertex = graph.adjList[i].vertex;
        while (edge) {
            degreeArray[vertex]++;
            // indegreeArray[edge->destination]++; // 入度
            edge = edge->next;
        }
    }
}

// 删除从v到w的边；
void deleteEdge(Graph* graph, int v, int w) {
    Edge* edge = NULL;
    for (int i = 0; i < graph->vertices; ++i) {
        if (graph.adjList[i].vertex == v) {
            edge = graph.adjList[i].head;
            Edge* prev = NULL;

            while (edge) {
                if (edge->destination == w) {
                    if (prev == NULL) {
                        graph->adjList[i].head = edge->next;
                    } else {
                        prev->next = edge->next;
                    }
                    free(edge);
                    return;
                }
                prev = edge;
                edge = edge->next;
    }
}

// 判断有向图是否有环
int dfs(DirectedGraph* graph, int v, int* visited, int* recStack) {
    visited[v] = 1;
    recStack[v] = 1;

    Edge* edge = graph->array[v].head;
    while (edge) {
        int dest = edge->destination;
        if (!visited[dest] && dfs(graph, dest, visited, recStack) || recStack[dest])
            return 1;
        edge = edge->next;
    }

    recStack[v] = 0;
    return 0;
}
int isCyclic(Graph* graph) {
    int* visited = (int*)malloc(graph->vertices * sizeof(int));
    int* recStack = (int*)malloc(graph->vertices * sizeof(int));
    for (int i = 0; i < graph->vertices; ++i) {
        visited[i] = 0;
        recStack[i] = 0;
    }

    for (int i = 0; i < graph->vertices; ++i) {
        if (!visited[i] && dfs(graph, i, visited, recStack))
            return 1;
    }

    free(visited);
    free(recStack);

    return 0;
}

int main() {
    return 0;
}
